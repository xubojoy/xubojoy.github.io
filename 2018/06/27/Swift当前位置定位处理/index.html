三、这里我建了一个类MapProcessor来专门处理有关地图定位

上代码：

class MapProcessor: NSObject {
    var locationManager : CLLocationManager?
    var location : CLLocation?
    static let shared = MapProcessor()
    //打开定位
    func loadLocation()
    {
        locationManager = CLLocationManager.init()
        locationManager?.desiredAccuracy = kCLLocationAccuracyBest
        locationManager?.distanceFilter = 10
        locationManager?.delegate = self
        locationManager?.requestWhenInUseAuthorization()
        locationManager?.startUpdatingLocation()
    }
}
实现代理CLLocationManagerDelegate方法
extension MapProcessor: CLLocationManagerDelegate{
    //获取定位信息
    func locationManager(_ manager: CLLocationManager, didUpdateLocations locations: [CLLocation]) {
        print(locations)
        let loc:CLLocation = locations[locations.count-1]
        location = locations.last
        if loc.horizontalAccuracy > 0 {
            let lat =  String.init(format: "%.1f", (location?.coordinate.latitude)!)
            let long = String.init(format: "%.1f", (location?.coordinate.longitude)!)
            print(String.init(format: "lat:%@,long:%@", lat,long))
//            location = CLLocation(latitude: 34.7472541716, longitude: 113.6249284647)
            locationManager?.stopUpdatingLocation()
            //将经纬度转换成城市名
            let geocoder:CLGeocoder = CLGeocoder()
            geocoder.reverseGeocodeLocation(location!, completionHandler: { (placemark, error) in
                if error == nil{
                    let array = placemark! as NSArray
                    let mark = array.firstObject as! CLPlacemark 
                    print("==获取定位==========",mark.addressDictionary!)
                    //城市
                    let city:String = (mark.addressDictionary! as NSDictionary).value(forKey: "City") as! String
                    //国家
                    let country:NSString = (mark.addressDictionary! as NSDictionary).value(forKey: "Country") as! NSString
                    //国家编码
                    let countryCode:NSString = (mark.addressDictionary! as NSDictionary).value(forKey: "CountryCode") as! NSString
                    //街道位置
                    let formattedAddressLines : NSString = ((mark.addressDictionary! as NSDictionary).value(forKey: "FormattedAddressLines") as AnyObject).firstObject as! NSString
                    //具体位置
                    let name:String = ((mark.addressDictionary! as NSDictionary).value(forKey: "Name") as! NSString) as String
                    //省
                    var state: String = ""
                    if (mark.addressDictionary! as NSDictionary).value(forKey: "State") != nil {
                        state = (mark.addressDictionary! as NSDictionary).value(forKey: "State") as! String
                    }else{
                        state = ""
                    } 
                    //区
                    let subLocality:NSString = (mark.addressDictionary! as NSDictionary).value(forKey: "SubLocality") as! NSString
//                    state, 省：%@,
                    print("=============",state)
                    print(String.init(format: "城市：%@,国家：%@,国家编码：%@,街道位置：%@,具体位置：%@,区：%@", city,country,countryCode,formattedAddressLines,name,subLocality))
                    AppStatus.shared.city = city.pregReplace(pattern: "市", with: "")
                    AppStatus.saveAppStatus()
                }else{
                    print(error!)
                }
            })
        }
    }
